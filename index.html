async function buildApp() { const userCode = document.getElementById('editor').value; // لێرەدا کلیلەکەت پارچە پارە دەکەین بۆ ئەوەی گیتھاب نەی سڕێتەوە // تێبینی: پیتی یەکەم و کۆتایی کلیلەکەت لێرە دابنێ (نموونە ghp_ + ئەوی تر) const p1 = "ghp_"; const p2 = "ئەو بەشەی تری کلیلەکەت لێرە دابنێ"; const cleanToken = p1 + p2; const repoOwner = "sasrashid-collab"; const repoName = "K-Factory"; // گۆڕینی دەقی دوگمەکە بۆ ئەوەی بەکارهێنەر بزانێت ئیش دەکات const btn = document.querySelector('.build-btn'); btn.innerText = "⏳ خەریکی ناردنە بۆ کارگە..."; btn.disabled = true; try { const response = await fetch(`https://api.github.com/repos/${repoOwner}/${repoName}/dispatches`, { method: 'POST', headers: { 'Authorization': `token ${cleanToken}`, 'Accept': 'application/vnd.github.v3+json' }, body: JSON.stringify({ event_type: 'build_apk', client_payload: { code: userCode } }) }); if (response.ok) { alert("✅ فەرمانەکە وەرگیرا! مەکینەی K_Store دەستی بەکار کرد. ئەپەکەت دوای ١٠ خولەک ئامادە دەبێت."); } else { alert("❌ کێشەیەک لە سێرڤەر هەیە. تکایە دواتر هەوڵ بدەرەوە."); } } catch (error) { alert("⚠️ پەیوەندی ئینتەرنێت جێگیر نییە."); } finally { btn.innerText = "🚀 دروستکردنی ئەپی APK"; btn.disabled = false; } }
